86 template <bool translate, bool divideByW, bool alignedOutput, int outputStep, bool alignedInput, int inputStep>
87 void Vector4f_SSETransformStreamGeneric(float* outputStream, const float* inputStream, unsigned count, const Matrix4x4f& transform)
88 {
89  union
90  {
91  Matrix4x4f m;
92  __m128 r[4];
93  } xform;
94 
95  // Using the union allows us to access the rows of the transform matrix as __m128 SSE values.
96  // Also the transform parameter might not be aligned to 128-bits, copying it to the union will fix that
97  // as it has to be 128-bit aligned.
98  xform.m = transform;
99  __m128 R0 = xform.r[0]; // R0 to R4 are the rows of the transform matrix
100  __m128 R1 = xform.r[1];
101  __m128 R2 = xform.r[2];
102  __m128 R3 = xform.r[3];
103  for (unsigned i = 0; i < count; ++i)
104  {
105  // Hint to the CPU that we will access the memory ahead and it should warm the cache to avoid a cache miss
106  // when we actually go to use it. A bit of tweaking was required to get the best value to look ahead by.
107  _mm_prefetch((const char*)&inputStream[256], _MM_HINT_T0);
108  __m128 vals;
109  if (alignedInput) // note these 'if's are only on template parameters so aren't in the generated code
110  {
111  vals = (__m128)_mm_load_si128((const __m128i *)inputStream);
112  }
113  else
114  {
115  vals = _mm_loadu_ps(inputStream);
116  }
117  // vals contains the next x,y,z entry from the input stream
118  // We now make a 128-bit X with x copied as x,x,x,x, and same for Y and Z.
119  __m128 X = _mm_shuffle_ps(vals,vals,_MM_SHUFFLE(0,0,0,0));
120  __m128 Y = _mm_shuffle_ps(vals,vals,_MM_SHUFFLE(1,1,1,1));
121  __m128 Z = _mm_shuffle_ps(vals,vals,_MM_SHUFFLE(2,2,2,2));
122  // We multiply Z with R2, so Z = z*R2[0], z*R2[1], z*R2[2], z*R2[3]
123  Z = _mm_mul_ps(Z,R2);
124  if (translate)
125  {
126  // We add R3, so Z = R3[0]+z*R2[0], R3[1]+z*R2[1], R3[2]+z*R2[2], R3[3]+z*R2[3]
127  Z = _mm_add_ps(Z,R3);
128  }
129  // We multiply Y with R1, so Y = y*R1[0], ...
130  Y = _mm_mul_ps(Y,R1);
131  // We add Y to Z, so Z = R3[0]+z*R2[0]+y*R1[0] ...
132  Z = _mm_add_ps(Z,Y);
133 
134  // TODO: add optional AVX2 optimizations
135  //_mm_fmadd_ps() // fused multiply+add instruction if have AVX2
136 
137  // We multiply X with R0, so X = x*R0[0], ...
138  X = _mm_mul_ps(X,R0);
139  // We add X to Z, so Z = R3[0] + z*R2[0] + y*R1[0] + x*R0[0], ...
140  Z = _mm_add_ps(Z,X);
141  // Z now contains our transformed point. For perspective transforms, we need to divide by W.
142  if (divideByW)
143  {
144  Z = _mm_div_ps(Z,_mm_shuffle_ps(Z,Z,_MM_SHUFFLE(3,3,3,3)));
145  }
146  if (alignedOutput)
147  {
148  _mm_stream_ps(outputStream,Z); // aligned and cache friendly
149  }
150  else
151  {
152  _mm_storeu_ps(outputStream,Z); // unaligned store
153  }
154  inputStream += inputStep;
155  outputStream += outputStep;
156  }
157 }

176 template <bool translate, bool divideByW, bool alignedOutput, int outputStep, bool alignedInput, int inputStep>
177 void Vector4f_TransformStreamGeneric(float* outputStream, const float* inputStream, unsigned count, const Matrix4x4f& transform)
178 {
179  for (unsigned i = 0; i < count; ++i)
180  {
181  Vector4f Z = Vector4f_Multiply(transform.row[2], Vector4f_Replicate(inputStream[2])); // z
182  if (translate)
183  {
184  Z = Vector4f_Add(transform.row[3], Z);
185  }
186  // This form of the transform function is tailored for platforms with fused multiply and add instructions
187  // and the compiler flags can be configured so that the compiler will be able to emit these.
188  // See the Vector4f_Transform function for the alternative form which transposes the transform matrix
189  // first and then uses dot products instead. On some platforms where there are optimizations for dot-products
190  // then this might instead be faster.
191  Z = Vector4f_Add(Z, Vector4f_Multiply(transform.row[1], Vector4f_Replicate(inputStream[1]))); // y
192  Z = Vector4f_Add(Z, Vector4f_Multiply(transform.row[0], Vector4f_Replicate(inputStream[0]))); // x
193  if (divideByW)
194  {
195  Z = Vector4f_Multiply(Z, Vector4f_Replicate(Scalar1f_One() / Z.w));
196  }
197  *(Vector4f*)outputStream = Z;
198  inputStream += inputStep;
199  outputStream += outputStep;
200  }
201 }

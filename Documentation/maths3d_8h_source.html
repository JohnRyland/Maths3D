46 //
47 // This code requires a modern C++ compiler but is mostly C and avoids using
48 // classes (with the exception of providing unit type safety discussed below).
49 //
50 // No header or library dependencies.
51 //
52 // The API attempts to be functional in style, with inputs and a returned
53 // output.
54 //
55 // The inputs are never mutated (modified) and the returned value is a newly
56 // constructed object. This is to maintain referential transparency. Some
57 // benefits to this approach are that it is easy to chain together calls to
58 // these APIs. It also means that they can be safely used in multithreaded
59 // code. From a performance point of view it assumes that the compiler will be
60 // able to perform RVO.
61 //
62 // Inputs are passed by const reference when the size of the input would be
63 // larger than the size of a pointer.
64 //
65 // Changing the scalar type will change the underlying value type used by
66 // vector and matrix.
67 //
68 // A scalar value used to represent an angle can sometimes be used incorrectly
69 // when an API is not clear if the units are degrees or radians. To avoid this,
70 // the API itself should document the units with a specific type that encode
71 // the units. The types Degrees and Radians below do this. These are actually
72 // classes and so this is C++, but it is worth the developer time it saves by
73 // ensuring there is no mistakes of this kind. The same could be used for other
74 // types of units as required, such as meters
75 // vs feet if the API were to need this.
76 //
77 // Both the vector and matrix type contain unions with a 'v' member which can
78 // be used for iteration of the components instead of needing to write code
79 // which requires copy paste of the same code for each component. This avoids
80 // mistakes, such as:
81 //
82 // Vector4f ret;
83 // ret.x = a.x + b.x;
84 // ret.y = a.y + b.y;
85 // ret.y = a.z + b.z;
86 //
87 // Do you see the mistake? Happens all the time. Far better is to write code
88 // like this:
89 //
90 // Vector4f ret;
91 // for (int i = 0; i < 3; ++i)
92 // ret.v[i] = a.v[i] + b.v[i];
93 //
94 // Less lines of code, less error prone, and easier to update the formula
95 // applied to each component.
96 //
97 // The API omits providing a 3 component wide vector and instead only provides
98 // a Vector4f. This is because when dealing with arrays of these, we will be
99 // able to optimize better the 4 wide version, particularly if it can be
100 // aligned to 128-bits and used with a SIMD optimized function to transform an
101 // array of them. For all the provided functions below, we don't bother
102 // providing SIMD optimizations as most compilers can auto-vectorize this code
103 // just fine and so it only will make the code more error prone, and not able
104 // to be re-compiled with different compiler flags to target different machine
105 // types or with or without different levels of SIMD support (eg MMX, SSE3, SSE4).
106 //
107 // For some special cases where performance is important, see the maths3d_ext.h
108 // file which contains SSE optimizations for transforming arrays of vectors. The
109 // intent of the _ext.h file is to put any code which is not part of the core
110 // API and is more bulky or platform specific.
111 //

312 inline Matrix4x4f Matrix4x4f_Set(const Scalar1f v[16])
313 {
314  Matrix4x4f ret;
315  for (int i = 0; i < 16; ++i)
316  ret.v[i] = v[i];
317  return ret;
318 }

320 inline Matrix4x4f Matrix4x4f_Zero()
321 {
322  Matrix4x4f ret;
323  for (int i = 0; i < 16; ++i)
324  ret.v[i] = Scalar1f_Zero();

329 {

342 static Matrix4x4f Matrix4x4f_Multiply(const Matrix4x4f& m1, const Matrix4x4f& m2)
343 {
344  Matrix4x4f ret = Matrix4x4f_Zero();
345  for (int i = 0; i < 4; i++)
346  for (int j = 0; j < 4; j++)
347  for (int k = 0; k < 4; k++)
348  ret.m[i][j] += m1.m[k][j] * m2.m[i][k];
349  return ret;
350 }
351 
352 static Matrix4x4f Matrix4x4f_Transposed(const Matrix4x4f& a)
353 {
354  Matrix4x4f ret;
355  for (int i = 0; i < 4; i++)
356  for (int j = 0; j < 4; j++)
357  ret.m[i][j] = a.m[j][i];
358  return ret;
359 }

361 static Matrix4x4f Matrix4x4f_Scaled(const Matrix4x4f& m, Scalar1f scale)
362 {
363  Matrix4x4f ret = m;
364  for (int i = 0; i < 16; i++)
365  ret.v[i] = m.v[i] * scale;
366  return ret;
367 }

369 static Matrix4x4f Matrix4x4f_ScaleXYZ(const Vector4f& scale)
370 {
371  Matrix4x4f ret = Matrix4x4f_Zero();
372  ret.m[0][0] = scale.x;
373  ret.m[1][1] = scale.y;
374  ret.m[2][2] = scale.z;
375  ret.m[3][3] = scale.w;
376  return ret;

379 // Helper for other functions
380 static Matrix4x4f Matrix4x4f_RotateCommon(Scalar1f angle, int axis)
381 {
382  const Scalar1f sx = ::sin(angle);
383  const Scalar1f cx = ::cos(angle);
384  Matrix4x4f ret = Matrix4x4f_Zero();
385  ret.m[0][2] = ret.m[1][0] = ret.m[2][1] = sx;
386  ret.m[0][1] = ret.m[1][2] = ret.m[2][0] = -sx;
387  ret.m[axis][0] = ret.m[axis][1] = ret.m[axis][2] = Scalar1f_Zero();
388  ret.m[0][axis] = ret.m[1][axis] = ret.m[2][axis] = Scalar1f_Zero();
389  ret.m[0][0] = ret.m[1][1] = ret.m[2][2] = cx;
390  ret.m[axis][axis] = ret.m[3][3] = Scalar1f_One();
391  return ret;
392 }

409 static Matrix4x4f Matrix4x4f_PerspectiveFrustum(Radians fieldOfView, Scalar1f aspectRatio, Scalar1f near, Scalar1f far)
410 {
411  const Scalar1f z = Scalar1f_Zero();
412  const Scalar1f one = Scalar1f_One();
413  const Scalar1f two = Scalar1f_Two();
414  const Scalar1f fov = fieldOfView.value;
415  const Scalar1f ctan = one / ::tanf(fov / two);
416  const Scalar1f inlf = one / (near - far);
417  const Scalar1f perspective[16] = { ctan / aspectRatio, z, z, z,
418  z, ctan, z, z,
419  z, z, (far + near) * inlf, -one,
420  z, z, (two * far * near) * inlf, z };
421  return Matrix4x4f_Set(perspective);
422 }

424 static Matrix4x4f Matrix4x4f_OrthographicFrustum(Scalar1f left, Scalar1f right, Scalar1f bottom, Scalar1f top, Scalar1f near, Scalar1f far)
425 {
426  const Scalar1f z = Scalar1f_Zero();
427  const Scalar1f two = Scalar1f_Two();
428  const Scalar1f width = right - left;
429  const Scalar1f height = top - bottom;
430  const Scalar1f depth = far - near;
431  const Scalar1f a = two / width;
432  const Scalar1f b = (right + left) / width;
433  const Scalar1f c = two / height;
434  const Scalar1f d = (top + bottom) / height;
435  const Scalar1f e = -(far + near) / depth;
436  Scalar1f ortho[16] = { a, z, z,-b,
437  z, c, z,-d,
438  z, z,-two/depth, e,
439  z, z, z, z };
440  return Matrix4x4f_Set(ortho);
441 }

443 static Vector4f Vector4f_Transform(const Matrix4x4f& m, const Vector4f& vec)
444 {
445  Matrix4x4f trans = Matrix4x4f_Transposed(m);
446  return Vector4f_Set(Vector4f_DotProduct(trans.row[0], vec),
447  Vector4f_DotProduct(trans.row[1], vec),
448  Vector4f_DotProduct(trans.row[2], vec),
449  Vector4f_DotProduct(trans.row[3], vec));
450 }

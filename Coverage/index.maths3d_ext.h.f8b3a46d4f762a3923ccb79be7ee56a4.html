159 /// Transforms arrays of vectors by the transform matrix (non-SSE fallback implementation, untested).
160 /// \tparam translate is a bool to enable or disable applying the translation component of the transform.
161 /// \tparam divideByW is a bool to enable or disable applying perspective by dividing by W.
162 /// \tparam alignedOutput is ignored
163 /// \tparam outputStep is the width in floats to the next vector element of the outputStream array.
164 /// This allows the outputStream array to be an array of structures of which the vector is a member.
165 /// These structures should be 32-bit aligned, and if the size is not a multiple of 128-bits, then
166 /// alignedOutput shouldn't be set.
167 /// \tparam alignedInput is ignored
168 /// \tparam inputStep is the width in floats to the next vector element of the inputStream array.
169 /// This allows the inputStream array to be an array of structures of which the vector is a member.
170 /// These structures should be 32-bit aligned, and if the size is not a multiple of 128-bits, then
171 /// alignedInput shouldn't be set.
172 /// \param outputStream is the output buffer to put the transformed vectors to.
173 /// \param inputStream is the input array of vectors to apply the transform to.
174 /// \param count is the number of vectors to transform.
175 /// \param transform is the matrix to apply.
176 template <bool translate, bool divideByW, bool alignedOutput, int outputStep, bool alignedInput, int inputStep>
177 void Vector4f_TransformStreamGeneric(float* outputStream, const float* inputStream, unsigned count, const Matrix4x4f& transform)
178 {
179 for (unsigned i = 0; i < count; ++i)
180 {
181 Vector4f Z = Vector4f_Multiply(transform.row[2], Vector4f_Replicate(inputStream[2])); // z
182 if (translate)
183 {
184 Z = Vector4f_Add(transform.row[3], Z);
185 }
186 // This form of the transform function is tailored for platforms with fused multiply and add instructions
187 // and the compiler flags can be configured so that the compiler will be able to emit these.
188 // See the Vector4f_Transform function for the alternative form which transposes the transform matrix
189 // first and then uses dot products instead. On some platforms where there are optimizations for dot-products
190 // then this might instead be faster.
191 Z = Vector4f_Add(Z, Vector4f_Multiply(transform.row[1], Vector4f_Replicate(inputStream[1]))); // y
192 Z = Vector4f_Add(Z, Vector4f_Multiply(transform.row[0], Vector4f_Replicate(inputStream[0]))); // x
193 if (divideByW)
194 {
195 Z = Vector4f_Multiply(Z, Vector4f_Replicate(Scalar1f_One() / Z.w));
196 }
197 *(Vector4f*)outputStream = Z;
198 inputStream += inputStep;
199 outputStream += outputStep;
200 }
